
**2.2. Создание конфигурации виртуального хоста**
```bash
# Создание файла конфигурации
cat > /etc/httpd2/conf/sites-available/example.com.conf << 'EOF'
<VirtualHost *:80>
    ServerAdmin admin@example.com
    ServerName example.com
    ServerAlias www.example.com
    DocumentRoot /var/www/example.com/public_html
    
    <Directory /var/www/example.com/public_html>
        Options -Indexes +FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>
    
</VirtualHost>
EOF

# Включение сайта
a2ensite example.com

# Отключение сайта по умолчанию (опционально)
a2dissite 000-default

# Проверка конфигурации и перезагрузка
apachectl configtest
systemctl reload httpd2
```

### Задание 4: Создание PHP-приложения

**4.1. Создание конфигурационного файла**
```bash
# Создание файла конфигурации
cat > /var/www/example.com/public_html/config.php << 'EOF'
<?php
// Конфигурация базы данных
define('DB_HOST', 'localhost');
define('DB_NAME', 'webapp');
define('DB_USER', 'webuser');
define('DB_PASS', 'SecurePassword123!');
define('DB_CHARSET', 'utf8mb4');

// Создание подключения через PDO
function getDBConnection() {
    try {
        $dsn = "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";charset=" . DB_CHARSET;
        $options = [
            PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES => false,
        ];
        return new PDO($dsn, DB_USER, DB_PASS, $options);
    } catch (PDOException $e) {
        throw new PDOException($e->getMessage(), (int)$e->getCode());
    }
}
?>
EOF

# Назначение прав (только для чтения владельцем)
chmod 640 /var/www/example.com/public_html/config.php
chown apache2:apache2 /var/www/example.com/public_html/config.php
```

**4.2. Создание страницы добавления пользователя**
```bash
cat > /var/www/example.com/public_html/add_user.php << 'EOF'
<?php
require_once 'config.php';

$message = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $username = trim($_POST['username'] ?? '');
    $email = trim($_POST['email'] ?? '');
    $password = $_POST['password'] ?? '';
    
    if (!empty($username) && !empty($email) && !empty($password)) {
        try {
            $pdo = getDBConnection();
            $stmt = $pdo->prepare("INSERT INTO users (username, email, password_hash) VALUES (?, ?, ?)");
            $password_hash = password_hash($password, PASSWORD_DEFAULT);
            $stmt->execute([$username, $email, $password_hash]);
            $message = "User created successfully!";
        } catch (PDOException $e) {
            $message = "Error: " . $e->getMessage();
        }
    } else {
        $message = "Please fill all fields!";
    }
}
?>
<!DOCTYPE html>
<html>
<head>
    <title>Add User</title>
</head>
<body>
    <h1>Add New User</h1>
    <?php if ($message): ?>
        <p><?php echo htmlspecialchars($message); ?></p>
    <?php endif; ?>
    <form method="POST">
        <label>Username: <input type="text" name="username" required></label><br>
        <label>Email: <input type="email" name="email" required></label><br>
        <label>Password: <input type="password" name="password" required></label><br>
        <button type="submit">Add User</button>
    </form>
    <p><a href="index.php">Back to Home</a> | <a href="users.php">View Users</a></p>
</body>
</html>
EOF
```

**4.3. Создание страницы просмотра пользователей**
```bash
cat > /var/www/example.com/public_html/users.php << 'EOF'
<?php
require_once 'config.php';

try {
    $pdo = getDBConnection();
    $stmt = $pdo->query("SELECT id, username, email, created_at FROM users ORDER BY id DESC");
    $users = $stmt->fetchAll();
} catch (PDOException $e) {
    $error = "Error: " . $e->getMessage();
}
?>
<!DOCTYPE html>
<html>
<head>
    <title>Users List</title>
</head>
<body>
    <h1>Users List</h1>
    <?php if (isset($error)): ?>
        <p><?php echo htmlspecialchars($error); ?></p>
    <?php elseif (!empty($users)): ?>
        <table border="1" cellpadding="10">
            <tr>
                <th>ID</th>
                <th>Username</th>
                <th>Email</th>
                <th>Created</th>
            </tr>
            <?php foreach ($users as $user): ?>
                <tr>
                    <td><?php echo htmlspecialchars($user['id']); ?></td>
                    <td><?php echo htmlspecialchars($user['username']); ?></td>
                    <td><?php echo htmlspecialchars($user['email']); ?></td>
                    <td><?php echo htmlspecialchars($user['created_at']); ?></td>
                </tr>
            <?php endforeach; ?>
        </table>
    <?php else: ?>
        <p>No users found.</p>
    <?php endif; ?>
    <p><a href="index.php">Back to Home</a> | <a href="add_user.php">Add User</a></p>
</body>
</html>
EOF
```

### Задание 5: Настройка безопасности

**5.1. Создание .htaccess файла**
```bash
cat > /var/www/example.com/public_html/.htaccess << 'EOF'
# Отключение листинга директорий
Options -Indexes

# Защита от XSS
<IfModule mod_headers.c>
    Header set X-XSS-Protection "1; mode=block"
    Header set X-Content-Type-Options "nosniff"
    Header set X-Frame-Options "SAMEORIGIN"
</IfModule>

# Ограничение загрузки файлов
<IfModule mod_php.c>
    php_value upload_max_filesize 10M
    php_value post_max_size 10M
</IfModule>

# Кэширование
<IfModule mod_expires.c>
    ExpiresActive On
    ExpiresByType text/css "access plus 1 month"
    ExpiresByType application/javascript "access plus 1 month"
    ExpiresByType image/jpeg "access plus 1 year"
    ExpiresByType image/png "access plus 1 year"
</IfModule>
EOF

# Назначение прав
chmod 644 /var/www/example.com/public_html/.htaccess
```

**5.2. Настройка базовой аутентификации для админ-панели**
```bash
# Создание директории админ-панели

EOF

# Создание .htpasswd файла

# Создание .htaccess для админ-панели

AuthType Basic
AuthName "Admin Area"
AuthUserFile /etc/apache2/.htpasswd
Require valid-user
EOF

# Назначение прав
chmod 640 /etc/httpd2/.htpasswd
chmod 644 /var/www/example.com/public_html/admin/.htaccess
```
